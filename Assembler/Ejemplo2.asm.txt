;Se crea una varible que se llama dato
; el 0x21 es la posicion en donde se almacena
; 0x21 proposito general

;CONFIGURACION
DATO EQU 0x21
;INICIO DE PROGRAMA
	ORG	0X00
	GOTO	START
;CODIGO
START
	BSF	STATUS,5
	CLRF	TRISB
	BCF	STATUS, RP0
	BCF	STATUS,5
	MOVLW	0X00
	MOVWF	PORTB
	GOTO	MENU
MENU
	BTFSS	PORTC, 7
	GOTO	IZQUIERDO
	GOTO	DERECHO
	GOTO	MENU
MV
	MOVF	DATO, W
	MOVWF	PORTB
	RETURN
UNO
	MOVLW	B'11111111'
	MOVWF	DATO
	CALL	MV
